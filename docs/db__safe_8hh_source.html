121 inline tchecker::dbm::db_t sum(tchecker::dbm::db_t const & db1, tchecker::dbm::db_t const & db2)
122 {
123  if ((db1 == tchecker::dbm::LT_INFINITY) || (db2 == tchecker::dbm::LT_INFINITY))
124  return tchecker::dbm::LT_INFINITY;
125  tchecker::integer_t sum = db1.value + db2.value;
126  if (db1.value > 0 && db2.value > 0 && sum < db1.value)
127  throw std::overflow_error("Overflow in sum of difference bounds");
128  if (db1.value < 0 && db2.value < 0 && sum > db1.value)
129  throw std::underflow_error("Underflow in sum of difference bounds");
130  return tchecker::dbm::db(tchecker::dbm::comparator_stronger(db1.cmp, db2.cmp), sum);
131 }
