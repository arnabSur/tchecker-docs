251 template <class REFZG, typename VEDGE_RANGE>
252 tchecker::refzg::path::finite_path_t<REFZG> * compute_run(std::shared_ptr<REFZG> const & refzg,
253  tchecker::vloc_t const & initial_vloc, VEDGE_RANGE const & seq)
254 {
255  tchecker::refzg::path::finite_path_t<REFZG> * path = new tchecker::refzg::path::finite_path_t<REFZG>{refzg};
256 
257  tchecker::refzg::const_state_sptr_t s{tchecker::refzg::initial(*refzg, initial_vloc)};
258  if (s.ptr() == nullptr) {
259  delete path;
260  throw std::invalid_argument("No initial state with given tuple of locations");
261  }
262 
263  path->add_first_node(s);
264  path->first()->initial(true);
265 
266  for (tchecker::const_vedge_sptr_t const & vedge_ptr : seq) {
267  s = path->last()->state_ptr();
268  auto && [nexts, nextt] = tchecker::refzg::next(*refzg, s, *vedge_ptr);
269  if (nexts.ptr() == nullptr || nextt.ptr() == nullptr) {
270  delete path;
271  throw std::invalid_argument("Sequence is not feasible from given initial locations");
272  }
273  path->extend_back(nextt, nexts);
274  }
275 
276  return path;
277 }
