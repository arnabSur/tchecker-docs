62 template <class GRAPH, class CEX> CEX * counter_example_zg(GRAPH const & g)
63 {
64  std::shared_ptr<tchecker::zg::zg_t> zg{
65  tchecker::zg::factory(g.zg().system_ptr(), tchecker::zg::STANDARD_SEMANTICS, tchecker::zg::NO_EXTRAPOLATION, 128, 128)};
66 
67  // compute sequence of edges from initial to final node in g
68  tchecker::algorithms::path::finite::algorithm_t<GRAPH> algorithm;
69 
70  std::vector<typename GRAPH::edge_sptr_t> seq =
71  algorithm.run(g, &tchecker::tck_reach::initial_node<GRAPH>, &tchecker::tck_reach::final_node<GRAPH>,
72  &tchecker::tck_reach::true_edge<GRAPH>);
73 
74  if (seq.empty())
75  return new CEX{zg};
76 
77  // Extract sequence of vedges
78  std::vector<tchecker::const_vedge_sptr_t> vedge_seq;
79  for (typename GRAPH::edge_sptr_t const & e : seq)
80  vedge_seq.push_back(e->vedge_ptr());
81 
82  // Get the corresponding run in a zone graph with standard semantics and no extrapolation
83  tchecker::vloc_t const & initial_vloc = g.edge_src(seq[0])->state().vloc();
84  CEX * cex = tchecker::zg::path::compute_run(zg, initial_vloc, tchecker::make_range(vedge_seq));
85  if (!cex->empty()) {
86  cex->first()->initial(true);
87  cex->last()->final(true);
