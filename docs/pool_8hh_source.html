149  template <class... ARGS> tchecker::intrusive_shared_ptr_t<T> construct(ARGS &&... args)
150  {
151  void * t = allocate();
152  if (t == nullptr)
153  return tchecker::intrusive_shared_ptr_t<T>(nullptr);
154  // p points after the reference counter
155  void * p = static_cast<typename T::refcount_t *>(t) + 1;
156  try {
157  T::construct(p, std::forward<ARGS>(args)...); // construct T in p with args
158  }
159  catch (...) {
160  this->release(t);
161  throw;
162  }
163  return tchecker::intrusive_shared_ptr_t<T>(reinterpret_cast<T *>(p));
164  }

176  bool destruct(tchecker::intrusive_shared_ptr_t<T> & p)
177  {
178  assert(p->refcount() >= 1);
179 
180  if (p.ptr() == nullptr)
181  return false;
182  if (p->refcount() > 1)
183  return false;
184 
185  T * t = p.ptr();
186  T::destruct(t);
187 
188  typename T::refcount_t * chunk = reinterpret_cast<typename T::refcount_t *>(t) - 1;
189  release(chunk);
190 
191  p = nullptr;
192 
193  return true;
194  }

203  std::size_t collect()
204  {
205  std::size_t collected = 0;
206  void *collected_begin = nullptr, *collected_end = nullptr;
207 
208  // Move to the free list all chunks in blocks list that are not in the
209  // free list and that are unused
210  for (void * block = _block_head; block != nullptr; block = nextblock(block)) {
211  void * block_end = static_cast<char *>(block) + _block_size;
212 
213  for (char * chunk = first_chunk_ptr(block); chunk != block_end; chunk += _alloc_size) {
214  // Ignore chunks inside unused raw block (refcount not set yet)
215  if ((_raw_head <= chunk) && (chunk < _raw_end))
216  break; // ignore the entire raw block
217 
218  // Collect all unused chunks in the free list
219  typename T::refcount_t * refcount = reinterpret_cast<typename T::refcount_t *>(chunk);
220  if (*refcount == COLLECTABLE_CHUNK) {
221  // make the chunk free using its refcount
222  *refcount = FREE_CHUNK;
223  // destruct the object in the chunk
224  T * t = reinterpret_cast<T *>(refcount + 1);
225  T::destruct(t); // t->~T()
226  // add the chunk to the list of collected chunks
227  this->nextchunk(chunk) = collected_begin;
228  if (collected_end == nullptr)
229  collected_end = chunk;
230  collected_begin = chunk;
231  ++collected;
232  }
233  }
234  }
235 
236  // release the collected chunks
237  if (collected > 0)
238  this->release(collected_begin, collected_end);
239 
240  return collected;
241  }

248  void destruct_all()
249  {
250  // Call destructor on all chunks in blocks list that
251  // - are not in free list
252  // - are not in raw block
253  for (void * block = _block_head; block != nullptr; block = nextblock(block)) {
254  void * block_end = static_cast<char *>(block) + _block_size;
255 
256  for (char * chunk = first_chunk_ptr(block); chunk != block_end; chunk += _alloc_size) {
257  // Ignore chunks inside unused raw block
258  if ((_raw_head <= chunk) && (chunk < _raw_end))
259  break;
260 
261  // Destruct all chunks that are not free
262  typename T::refcount_t * refcount = reinterpret_cast<typename T::refcount_t *>(chunk);
263 
264  if (*refcount > T::REFCOUNT_MAX)
265  continue;
266

268  // make the chunk free using its refcount
269  *refcount = FREE_CHUNK;
270  // destruct the object in the chunk
271  T * t = reinterpret_cast<T *>(refcount + 1);
272  T::destruct(t); // t->~T();
