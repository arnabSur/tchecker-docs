83  shared_A_t * a2 = shared_A_t::allocate_and_construct(1, 2);
84  tchecker::intrusive_shared_ptr_t<shared_A_t> p2(a2);

86  tchecker::intrusive_shared_ptr_t<shared_A_t> p3(a3);
87 
88  SECTION("Caching an A")
89  {
90  REQUIRE(cache.find_else_add(p1) == p1);
91  REQUIRE(cache.find(p1));
92  REQUIRE(cache.size() == 1);
93  }
94 
95  SECTION("Finding a cached A")
96  {
97  cache.find_else_add(p1);
98  REQUIRE(cache.find(p1));
99 
100  REQUIRE(cache.find_else_add(p1bis) == p1);
101  REQUIRE(cache.size() == 1);
102  }
103 
104  SECTION("Not finding a non-cached A")
105  {
106  cache.find_else_add(p1);
107  REQUIRE(cache.find(p1));
108 
109  REQUIRE_FALSE(cache.find(p2));

111 
112  REQUIRE_FALSE(cache.find(p3));
113  REQUIRE(cache.size() == 1);
114  }
115
