458  template <class TAG> void remove_edge(EDGE_PTR const & e)
459  {
460  NODE_PTR n = e->template node<TAG>();
461  auto range = edges<TAG>(n);
462  auto end = range.end();
463  EDGE_PTR prev{nullptr}, next{nullptr};
464 
465  for (auto it = range.begin(); it != end; ++it) {
466  if (*it == e) {
467  // get next edge in list (if any)
468  ++it;
469  if (it != end)
470  next = *it;
471  else // no next edge
472  next = EDGE_PTR{nullptr};
473 
474  // remove edge
475  if (prev == EDGE_PTR{nullptr}) // first edge in list
476  n->template head<TAG>() = next;
477  else
478  prev->template next<TAG>() = next;
479 
480  e->template next<TAG>() = EDGE_PTR{nullptr};
481  e->template node<TAG>() = NODE_PTR{nullptr};
482 
483  return;
484  }
485  prev = *it;
486  }
