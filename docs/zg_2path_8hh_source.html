250 template <class ZG, typename VEDGE_RANGE>
251 tchecker::zg::path::finite_path_t<ZG> * compute_run(std::shared_ptr<ZG> const & zg, tchecker::vloc_t const & initial_vloc,
252  VEDGE_RANGE const & seq)
253 {
254  tchecker::zg::path::finite_path_t<ZG> * path = new tchecker::zg::path::finite_path_t<ZG>{zg};
255 
256  tchecker::zg::const_state_sptr_t s{tchecker::zg::initial(*zg, initial_vloc)};
257  if (s.ptr() == nullptr) {
258  delete path;
259  throw std::invalid_argument("No initial state with given tuple of locations");
260  }
261 
262  path->add_first_node(s);
263  path->first()->initial(true);
264 
265  for (tchecker::const_vedge_sptr_t const & vedge_ptr : seq) {
266  s = path->last()->state_ptr();
267  auto && [nexts, nextt] = tchecker::zg::next(*zg, s, *vedge_ptr);
268  if (nexts.ptr() == nullptr || nextt.ptr() == nullptr) {
269  delete path;
270  throw std::invalid_argument("Sequence is not feasible from given initial locations");
271  }
272  path->extend_back(nextt, nexts);
273  }
274 
275  return path;
276 }
