548 tchecker::refzg::state_sptr_t initial(REFZG & refzg, tchecker::vloc_t const & vloc,
549  tchecker::state_status_t mask = tchecker::STATE_OK)
550 {
551  std::vector<typename REFZG::sst_t> v;
552  refzg.initial(v, mask);
553  for (auto && [status, s, t] : v) {
554  if (s->vloc() == vloc)
555  return s;
556  }
557  return nullptr;
558 }

569 template <class REFZG>
570 std::tuple<tchecker::refzg::state_sptr_t, tchecker::refzg::transition_sptr_t>
571 next(REFZG & refzg, tchecker::refzg::const_state_sptr_t const & s, tchecker::vedge_t const & vedge,
572  tchecker::state_status_t mask = tchecker::STATE_OK)
573 {
574  std::vector<typename REFZG::sst_t> v;

576  for (auto && [status, nexts, nextt] : v)
