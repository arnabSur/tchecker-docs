544  void remove(tchecker::collision_table_position_t position_in_table,
545  tchecker::collision_table_position_t position_in_collision_list)
546  {
547  if (position_in_table >= _table.size())
548  throw std::invalid_argument("Removing an object which is not stored in this collision table");
549 
550  collision_list_t & c = _table[position_in_table];
551 
552  if (position_in_collision_list >= c.size())
553  throw std::invalid_argument("Removing an object that is not stored in this collision list");
554 
555  c[position_in_collision_list]->clear_position();
556 
557  if (position_in_collision_list < c.size() - 1) {
558  SPTR back_object = c.back();
559  back_object->clear_position();
560  back_object->set_position(position_in_table, position_in_collision_list);
561  c[position_in_collision_list] = back_object;
562  }
563 
564  c.pop_back();
565  --_size;
566  }
