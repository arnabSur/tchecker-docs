202  void merge_scc(node_sptr_t & n, tchecker::algorithms::couvscc::stats_t & stats, boost::dynamic_bitset<> const & labels)
203  {
204  boost::dynamic_bitset<> scc_labels{labels.size(), 0}, ulabels{labels.size(), 0};
205  node_sptr_t u{nullptr};
206  do {
207  u = _roots.top().n;
208  scc_labels |= _roots.top().labels;
209  if (!labels.none() && labels.is_subset_of(scc_labels)) {
210  stats.cycle() = true;
211  return;
212  }
213  _roots.pop();
214  } while (u->dfsnum() > n->dfsnum());
215  _roots.push(roots_stack_entry_t{u, scc_labels});

250  std::deque<node_sptr_t> expand_node(node_sptr_t & n, TS & ts, GRAPH & graph, tchecker::algorithms::couvscc::stats_t & stats)
251  {
252  std::deque<node_sptr_t> next_nodes;
253  std::vector<typename TS::sst_t> v;
254  ts.next(n->state_ptr(), v);
255  for (auto && [status, s, t] : v) {
256  ++stats.visited_transitions();
