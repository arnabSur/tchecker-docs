115  std::vector<typename TS::sst_t> sst;
116  ts.initial(sst);
117  for (auto && [status, s, t] : sst) {
118  auto && [is_new_node, initial_node] = graph.add_node(s);
119  initial_node->initial(true);

187  void dfs_blue(TS & ts, GRAPH & graph, boost::dynamic_bitset<> const & labels, tchecker::algorithms::ndfs::stats_t & stats,
188  node_sptr_t & n)
189  {
190  std::stack<blue_stack_entry_t> stack;
191 
192  n->color() = tchecker::algorithms::ndfs::CYAN;
193  stack.push(blue_stack_entry_t{n, expand_node(ts, graph, n), true});
194  ++stats.visited_states_blue();
195 
196  while (!stack.empty()) {
197  auto && [s, succ, allred] = stack.top();
198  if (succ.empty()) {
199  if (allred)
200  s->color() = tchecker::algorithms::ndfs::RED;
201  else if (accepting(s, ts, labels)) {
202  dfs_red(ts, graph, labels, stats, s);
203  s->color() = tchecker::algorithms::ndfs::RED;
204  }
205  else
206  s->color() = tchecker::algorithms::ndfs::BLUE;
207  bool s_is_red = (s->color() == tchecker::algorithms::ndfs::RED);
208  stack.pop();
209  if (!s_is_red && !stack.empty())
210  stack.top().allred = false;
211  }
212  else {
213  node_sptr_t t = stack.top().pick_successor();
214  ++stats.visited_transitions_blue();
215  if (t->color() == tchecker::algorithms::ndfs::CYAN && (accepting(s, ts, labels) || accepting(t, ts, labels))) {
216  stats.cycle() = true;
217  break;
218  }
219  else if (t->color() == tchecker::algorithms::ndfs::WHITE) {
220  t->color() = tchecker::algorithms::ndfs::CYAN;
221  stack.push(blue_stack_entry_t{t, expand_node(ts, graph, t), true});
222  ++stats.visited_states_blue();
223  }
224  else if (t->color() != tchecker::algorithms::ndfs::RED)
225  allred = false;
226  }
227  }
228  }
